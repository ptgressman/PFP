\begin{problem}
  Set up the definite integral
    \[
      \int_{\answer{0}}^{\answer{1}} f(x) \, dx
    \]
  to evaluate from $0$ to $1$.

  \begin{question}
    What if $f(x) = \sin x$?  Then,
    \[
       \int_0^1 \sin x \, dx = \answer{-\cos(1) + \cos(0)}.
    \]

    \begin{hint}
      Recall $\int \sin x \, dx = -\cos x + C$.
    \end{hint}
  \end{question}
\end{problem}
